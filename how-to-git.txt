// identity
git config --global user.name "..."
git config --global user.email "..."

// change editor (default is vi)
git config --global core.editor nano

// list config
git config --list

// init git at current dir
git init

// add everything to git
git add .

// commit - this opens the editor to add a commit message (no message no commit)
git commit

// list remote
git remote -v

// add remote
git remote add origin https://github.com/adleiseifer/fun.git

// change url
git remote set-url origin https://github.com/adleiseifer/fun.git

// clone
git clone https://github.com/adleiseifer/fun.git
